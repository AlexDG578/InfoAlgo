NR DIVIZORI
#include <iostream>
using namespace std;

int n,d,x;
int main()
{
   cin>>n;
   for(x = 1 ; x * x <= n ; x++)
    if(n % x == 0)
   {
       d++;
       if(n / x > x)
        d++;
   }
   cout<<d;
}

SUMA DIVIZORI
#include <iostream>
using namespace std;

int n,d,x,s;
int main()
{
   cin>>n;
   for(x = 1 ; x * x <= n ; x++)
    if(n % x == 0)
   {
       s = s + x;
       if(n / x > x)
        s = s + n / x;
   }
   cout<<s;
}
